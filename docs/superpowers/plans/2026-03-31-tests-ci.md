### Task 3: Write transcript unit tests

**Files:**
- Create: `apps/ionosphere/src/lib/transcript.test.ts`

- [ ] **Step 1: Write the test file**

```typescript
import { describe, it, expect } from "vitest";
import {
  extractData,
  brightnessAtTime,
  toColor,
  BASE_BRIGHTNESS,
  PEAK_BRIGHTNESS,
  WINDOW_NS,
  type TranscriptDocument,
} from "./transcript";

// --- Test fixtures ---

function makeDoc(words: Array<{ text: string; startNs: number; endNs: number }>, concepts?: Array<{ byteStart: number; byteEnd: number; name: string }>): TranscriptDocument {
  const encoder = new TextEncoder();
  const text = words.map((w) => w.text).join(" ");
  const facets: TranscriptDocument["facets"] = [];

  let searchFrom = 0;
  for (const w of words) {
    const idx = text.indexOf(w.text, searchFrom);
    const byteStart = encoder.encode(text.slice(0, idx)).length;
    const byteEnd = encoder.encode(text.slice(0, idx + w.text.length)).length;

    facets.push({
      index: { byteStart, byteEnd },
      features: [{
        $type: "tv.ionosphere.facet#timestamp",
        startTime: w.startNs,
        endTime: w.endNs,
      }],
    });
    searchFrom = idx + w.text.length;
  }

  if (concepts) {
    for (const c of concepts) {
      facets.push({
        index: { byteStart: c.byteStart, byteEnd: c.byteEnd },
        features: [{
          $type: "tv.ionosphere.facet#concept-ref",
          conceptUri: `at://test/tv.ionosphere.concept/${c.name}`,
          conceptRkey: c.name,
          conceptName: c.name,
        }],
      });
    }
  }

  return { text, facets };
}

// --- extractData ---

describe("extractData", () => {
  it("extracts words sorted by start time", () => {
    const doc = makeDoc([
      { text: "Hello", startNs: 0, endNs: 500_000_000 },
      { text: "world", startNs: 500_000_000, endNs: 1_000_000_000 },
    ]);

    const { words } = extractData(doc);
    expect(words).toHaveLength(2);
    expect(words[0].text).toBe("Hello");
    expect(words[1].text).toBe("world");
    expect(words[0].startTime).toBe(0);
    expect(words[1].startTime).toBe(500_000_000);
  });

  it("computes shared boundary times between adjacent words", () => {
    const doc = makeDoc([
      { text: "Hello", startNs: 0, endNs: 400_000_000 },
      { text: "world", startNs: 600_000_000, endNs: 1_000_000_000 },
    ]);

    const { words } = extractData(doc);

    // First word: boundaryStart = own startTime, boundaryEnd = midpoint to next
    expect(words[0].boundaryStartTime).toBe(0);
    expect(words[0].boundaryEndTime).toBe(500_000_000); // (400M + 600M) / 2

    // Second word: boundaryStart = midpoint from prev, boundaryEnd = own endTime
    expect(words[1].boundaryStartTime).toBe(500_000_000);
    expect(words[1].boundaryEndTime).toBe(1_000_000_000);

    // KEY INVARIANT: word N boundaryEnd === word N+1 boundaryStart
    expect(words[0].boundaryEndTime).toBe(words[1].boundaryStartTime);
  });

  it("matches concepts to words by byte range overlap", () => {
    const encoder = new TextEncoder();
    // "AT Protocol" — concept covers the whole phrase
    const doc = makeDoc([
      { text: "AT", startNs: 0, endNs: 200_000_000 },
      { text: "Protocol", startNs: 200_000_000, endNs: 600_000_000 },
      { text: "rocks", startNs: 700_000_000, endNs: 1_000_000_000 },
    ]);

    const text = "AT Protocol rocks";
    const conceptStart = 0;
    const conceptEnd = encoder.encode("AT Protocol").length;

    doc.facets.push({
      index: { byteStart: conceptStart, byteEnd: conceptEnd },
      features: [{
        $type: "tv.ionosphere.facet#concept-ref",
        conceptUri: "at://test/tv.ionosphere.concept/at-protocol",
        conceptRkey: "at-protocol",
        conceptName: "AT Protocol",
      }],
    });

    const { wordConcepts } = extractData(doc);

    // "AT" and "Protocol" overlap with the concept
    expect(wordConcepts[0]).toHaveLength(1);
    expect(wordConcepts[0][0].conceptName).toBe("AT Protocol");
    expect(wordConcepts[1]).toHaveLength(1);
    // "rocks" does not
    expect(wordConcepts[2]).toHaveLength(0);
  });

  it("handles empty document", () => {
    const doc: TranscriptDocument = { text: "", facets: [] };
    const { words, concepts, wordConcepts } = extractData(doc);
    expect(words).toHaveLength(0);
    expect(concepts).toHaveLength(0);
    expect(wordConcepts).toHaveLength(0);
  });
});

// --- brightnessAtTime ---

describe("brightnessAtTime", () => {
  it("returns peak brightness at current time", () => {
    expect(brightnessAtTime(1_000_000_000, 1_000_000_000)).toBe(PEAK_BRIGHTNESS);
  });

  it("returns base brightness beyond the window", () => {
    const current = 5_000_000_000;
    const distant = current + WINDOW_NS + 1;
    expect(brightnessAtTime(current, distant)).toBe(BASE_BRIGHTNESS);
  });

  it("returns intermediate brightness within the window", () => {
    const current = 5_000_000_000;
    const halfway = current + WINDOW_NS / 2;
    const b = brightnessAtTime(current, halfway);
    expect(b).toBeGreaterThan(BASE_BRIGHTNESS);
    expect(b).toBeLessThan(PEAK_BRIGHTNESS);
  });

  it("is symmetric around current time", () => {
    const current = 5_000_000_000;
    const offset = 500_000_000;
    expect(brightnessAtTime(current, current + offset))
      .toBe(brightnessAtTime(current, current - offset));
  });

  it("uses quadratic easing (not linear)", () => {
    const current = 5_000_000_000;
    const quarter = current + WINDOW_NS / 4;
    const half = current + WINDOW_NS / 2;
    const bQuarter = brightnessAtTime(current, quarter);
    const bHalf = brightnessAtTime(current, half);

    // With quadratic easing, the drop from peak to quarter should be
    // less than the drop from quarter to half (steeper falloff further out)
    const dropToQuarter = PEAK_BRIGHTNESS - bQuarter;
    const dropQuarterToHalf = bQuarter - bHalf;
    expect(dropQuarterToHalf).toBeGreaterThan(dropToQuarter);
  });
});

// --- toColor ---

describe("toColor", () => {
  it("returns grayscale for non-concept words", () => {
    const color = toColor(0.5, null);
    expect(color).toMatch(/^rgb\(\d+ \d+ \d+\)$/);
    // Grayscale: all three channels equal
    const [r, g, b] = color.match(/\d+/g)!.map(Number);
    expect(r).toBe(g);
    expect(r).toBe(b);
  });

  it("returns amber tint for concept words", () => {
    const concept = {
      byteStart: 0,
      byteEnd: 5,
      conceptUri: "at://test/concept/foo",
      conceptRkey: "foo",
      conceptName: "Foo",
    };
    const color = toColor(0.8, concept);
    const [r, g, b] = color.match(/\d+/g)!.map(Number);
    // Amber: red > green > blue
    expect(r).toBeGreaterThan(g);
    expect(g).toBeGreaterThan(b);
  });

  it("returns near-gray for dim concepts", () => {
    const concept = {
      byteStart: 0,
      byteEnd: 5,
      conceptUri: "at://test/concept/foo",
      conceptRkey: "foo",
      conceptName: "Foo",
    };
    const color = toColor(BASE_BRIGHTNESS, concept);
    const [r, g, b] = color.match(/\d+/g)!.map(Number);
    // At base brightness, the amber tint should be very subtle
    // (channels close together)
    expect(Math.abs(r - b)).toBeLessThan(30);
  });
});
```

## Chunk 2: CI/CD Pipeline

### Task 4: Create GitHub Actions workflow

**Files:**
- Create: `.github/workflows/ci.yml`

- [ ] **Step 1: Create the workflow file**

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  check:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: 10

      - uses: actions/setup-node@v4
        with:
          node-version: 24
          cache: pnpm

      - run: pnpm install --frozen-lockfile

      - name: Typecheck
        run: pnpm -r typecheck

      - name: Test
        run: pnpm -r test
```

Note: `pnpm -r typecheck` runs `tsc --noEmit` in format and appview packages. The frontend doesn't have a typecheck script yet — add one if it fails, but Next.js projects typically typecheck during `next build`.
